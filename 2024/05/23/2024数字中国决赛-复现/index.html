app.set('view engine', 'ejs');设置了渲染引擎为ejs

然后就是三个路由

app.get('/', (req, res, next) => {
    res.redirect('/home?page=Home');
});

app.all('/home', function(req, res) {

app.all('/about', function(req, res) {
